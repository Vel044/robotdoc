```Python
@safe_stop_image_writer
def record_loop(
    robot: Robot,
    events: dict,
    fps: int,
    dataset: LeRobotDataset | None = None,
    teleop: Teleoperator | list[Teleoperator] | None = None,
    policy: PreTrainedPolicy | None = None,
    control_time_s: int | None = None,
    single_task: str | None = None,
    display_data: bool = False,
):
    """
    核心控制循环：负责观测获取、动作推理、指令发送及数据保存。
    该循环体现了：观测 -> 推理 -> 动作 -> 执行 -> 等待 的完整链路。
    """
    # 检查数据集帧率与请求帧率是否匹配，确保控制频率一致性
    if dataset is not None and dataset.fps != fps:
        raise ValueError(f"The dataset fps should be equal to requested fps ({dataset.fps} != {fps}).")

    # 初始化遥操作设备变量（支持机械臂和键盘组合控制）
    teleop_arm = teleop_keyboard = None
    if isinstance(teleop, list):
        # 筛选列表中的键盘控制对象和机械臂控制对象
        teleop_keyboard = next((t for t in teleop if isinstance(t, KeyboardTeleop)), None)
        teleop_arm = next(
            (
                t
                for t in teleop
                if isinstance(t, (so100_leader.SO100Leader, so101_leader.SO101Leader, koch_leader.KochLeader))
            ),
            None,
        )

        # 校验多模态遥操作的合法性（目前仅支持 LeKiwi 机器人）
        if not (teleop_arm and teleop_keyboard and len(teleop) == 2 and robot.name == "lekiwi_client"):
            raise ValueError(
                "For multi-teleop, the list must contain exactly one KeyboardTeleop and one arm teleoperator. Currently only supported for LeKiwi robot."
            )

    # 如果提供了策略模型（如 ACT），在开始前重置其内部状态（如 Transformer 的缓存）
    if policy is not None:
        policy.reset()

    timestamp = 0
    start_episode_t = time.perf_counter() # 记录回合开始的绝对时间

    # 主循环：在指定的回合时间内持续运行
    while timestamp < control_time_s:
        start_loop_t = time.perf_counter() # 记录本帧循环的开始时间

        # 检查是否通过键盘等事件触发了提前退出
        if events["exit_early"]:
            events["exit_early"] = False
            break

        # --- ① 获取观测 (Observation) ---
        # 链路：机器人硬件 -> 内核驱动缓冲区 -> Python dict
        observation = robot.get_observation()

        # 如果需要保存数据或执行策略，将原始观测转换为数据集定义的特征格式
        if policy is not None or dataset is not None:
            observation_frame = build_dataset_frame(dataset.features, observation, prefix="observation")

        # --- ② 推理/获取动作 (Action Generation) ---
        if policy is not None:
            # 自主模式：调用神经网络模型（如 ACT）预测动作值
            action_values = predict_action(
                observation_frame,
                policy,
                get_safe_torch_device(policy.config.device),
                policy.config.use_amp,
                task=single_task,
                robot_type=robot.robot_type,
            )
            # 将张量结果映射回机器人的动作特征键值对
            action = {key: action_values[i].item() for i, key in enumerate(robot.action_features)}
   
        # --- ③ 动作发送 (Send Action) ---
        # 链路：Python -> os.write -> TTY/ACM -> USB Controller (DMA) -> 机器人控制器
        # 注意：此处 send_action 在数据提交给内核/硬件队列后即返回，与物理传输并行
        sent_action = robot.send_action(action)

        # --- 数据记录 (Logging) ---
        if dataset is not None:
            # 构建动作帧并将其与观测帧合并，存入数据集缓冲区
            action_frame = build_dataset_frame(dataset.features, sent_action, prefix="action")
            frame = {**observation_frame, **action_frame}
            dataset.add_frame(frame, task=single_task)

        if display_data:
            # 如果开启可视化，将数据推送到 Rerun 等工具
            log_rerun_data(observation, action)

        # --- ④ 频率对齐 (Busy Wait) ---
        # 计算本轮逻辑消耗的时间
        dt_s = time.perf_counter() - start_loop_t
        # 忙等：等待直到填满 1/fps 的时长，确保控制频率稳定，--dataset.fps int：Limit the frames per second. (default: 30)，默认 FPS 是 30。
        # 此时硬件正在通过 DMA 并行发送动作包
        busy_wait(1 / fps - dt_s)

        # 更新回合已持续的时间
        timestamp = time.perf_counter() - start_episode_t
```

```python
def predict_action(
    observation: dict[str, np.ndarray],
    policy: PreTrainedPolicy,
    device: torch.device,
    use_amp: bool,
    task: str | None = None,
    robot_type: str | None = None,
):
    # 浅拷贝观测数据，避免修改原始数据源
    observation = copy(observation)
    
    with (
        # 开启推理模式：禁用梯度计算，减少内存消耗并加速计算
        torch.inference_mode(),
        # 如果使用 CUDA 且开启了混合精度(AMP)，则启用自动混合精度上下文
        torch.autocast(device_type=device.type) if device.type == "cuda" and use_amp else nullcontext(),
    ):
        # 遍历所有观测特征（如图像、关节角度等），将其转换为 PyTorch 格式
        for name in observation:
            # 将 NumPy 数组转换为 PyTorch 张量
            observation[name] = torch.from_numpy(observation[name])
            
            # 针对图像数据的特殊预处理：
            if "image" in name:
                # 归一化：将像素值从 [0, 255] 映射到 [0, 1.0] 的 float32 类型
                observation[name] = observation[name].type(torch.float32) / 255
                # 维度转换：将 (H, W, C) 转换为 PyTorch 要求的 (C, H, W) 格式
                observation[name] = observation[name].permute(2, 0, 1).contiguous()
            
            # 增加批次维度 (Batch Dimension)，将形状变为 (1, C, H, W)
            observation[name] = observation[name].unsqueeze(0)
            # 将数据移动到计算设备（如 CPU、CUDA 或 MPS）
            observation[name] = observation[name].to(device)

        # 注入任务描述和机器人类型信息（用于多任务策略）
        observation["task"] = task if task else ""
        observation["robot_type"] = robot_type if robot_type else ""

        # --- 核心推理阶段 ---
        # 调用策略模型的 select_action 方法，根据当前观测计算动作
        # 对于 ACT 模型，这会通过 Transformer 的编码器-解码器生成动作序列
        action = policy.select_action(observation)

        # 移除推理结果中的批次维度
        action = action.squeeze(0)

        # 将生成的动作移动回 CPU 内存，以便后续通过串口(os.write)发送给硬件
        action = action.to("cpu")

    return action
```

```c
// 实现
Py_ssize_t
_Py_write(int fd, const void *buf, size_t count)
{
    _Py_AssertHoldsTstate();

    /* _Py_write() must not be called with an exception set, otherwise the
     * caller may think that write() was interrupted by a signal and the signal
     * handler raised an exception. */
    assert(!PyErr_Occurred());

    return _Py_write_impl(fd, buf, count, 1);
}
// _Py_write_impl()函数的实现，这里的代码把MS_Windows的实现删了，只有Linux的
static Py_ssize_t
_Py_write_impl(int fd, const void *buf, size_t count, int gil_held)
{
    Py_ssize_t n;
    int err;
    int async_err = 0;

    // 限制写入最大长度
    if (count > _PY_WRITE_MAX) {
        count = _PY_WRITE_MAX;
    }

    if (gil_held) {
        // 持有 GIL（Global Interpreter Lock，全局解释器锁，是cpython的一种线程管理机制） 的路径：会释放 GIL，GIL是CPython 内部机制（管理解释器锁）
        do {
            Py_BEGIN_ALLOW_THREADS
            errno = 0;
            n = wrzhijieite(fd, buf, count);    // ← Linux 的 write 调用，这一行很重要，Linux 系统调用入口，由 Glibc 实现包装，最终进入内核 sys_write
            err = errno;                  // errno 来自 Linux 内核的系统调用返回机制，由内核设置，通过 glibc 传递到用户态，有内核错误的话就会收到内核错误码
            Py_END_ALLOW_THREADS
        } while (n < 0 &&
                 err == EINTR &&          //EINTR 重试（CPython 控制逻辑，OS 不负责）
                 !(async_err = PyErr_CheckSignals()));
    }
    else {
        // 未持有 GIL 时不检查 Python 信号
        do {
            errno = 0;
            n = write(fd, buf, count);
            err = errno;
        } while (n < 0 && err == EINTR);
    }

    // 遇到 Python signal handler 设置异常
    if (async_err) {
        errno = err;
        return -1;
    }

    // 遇到普通错误
    if (n < 0) {
        if (gil_held)
            PyErr_SetFromErrno(PyExc_OSError);
        errno = err;
        return -1;
    }

    return n;
}
```

```c
ssize_t ksys_write(unsigned int fd, const char __user *buf, size_t count)
{
	struct fd f = fdget_pos(fd);
	ssize_t ret = -EBADF;

	if (fd_file(f)) {
		loff_t pos, *ppos = file_ppos(fd_file(f));
		if (ppos) {
			pos = *ppos;
			ppos = &pos;
		}
		ret = vfs_write(fd_file(f), buf, count, ppos);  // 调用 vfs_write
		if (ret >= 0 && ppos)
			fd_file(f)->f_pos = pos;
		fdput_pos(f);
	}

	return ret;
}
```

#### 3.3. vfs_write

```c
/*
 * vfs_write - VFS 写入入口
 * @file:  目标文件
 * @buf:   用户态缓冲区指针
 * @count: 请求写入的字节数
 * @pos:   文件偏移（为 NULL 时使用 file->f_pos）
 *
 * 对用户写操作进行通用检查、限制和调度：
 * - 权限与可写性验证
 * - 用户空间指针有效性检查
 * - 文件边界与追加模式处理
 * - 写事务开始与结束（文件系统可能使用）
 * - 分发到具体文件系统/设备的 write/write_iter 实现
 */
ssize_t vfs_write(struct file *file, const char __user *buf,
                  size_t count, loff_t *pos)
{
    ssize_t ret;

    /* 验证文件是否以可写方式打开（O_WRONLY/O_RDWR） */
    if (!(file->f_mode & FMODE_WRITE))
        return -EBADF;

    /* 文件系统层面的可写性检查（例如只读挂载） */
    if (!(file->f_mode & FMODE_CAN_WRITE))
        return -EINVAL;

    /* 检查用户态地址 buf 是否可访问（防止越界或无效指针） */
    if (unlikely(!access_ok(buf, count)))
        return -EFAULT;

    /*
     * 检查文件锁、大小限制、O_APPEND 模式下的偏移调整。
     * 若超出允许范围或违反锁规则会直接失败。
     */
    ret = rw_verify_area(WRITE, file, pos, count);
    if (ret)
        return ret;

    /* 限制单次写入量，避免极端大写入（一般是 2GB 左右的上限） */
    if (count > MAX_RW_COUNT)
        count = MAX_RW_COUNT;

    /*
     * 通知文件系统开始一次写事务。
     * 对 journaling FS（如 ext4, btrfs）可能开启日志事务。
     */
    file_start_write(file);

    /*
     * 调度到具体的文件操作方法：
     *  1. file_operations->write       旧接口（不推荐）
     *  2. file_operations->write_iter  新接口（主流文件系统使用）
     */
    if (file->f_op->write)                                  // 调用具体文件系统的 write/write_iter
        ret = file->f_op->write(file, buf, count, pos);     
    else if (file->f_op->write_iter)
        ret = new_sync_write(file, buf, count, pos);        // 调用 new_sync_write，最终调用 write_iter
    else
        ret = -EINVAL;

    /*
     * 若写入成功：
     * - 触发 fsnotify（通知 inotify/fanotify）
     * - 将写入字节记入当前进程统计
     */
    if (ret > 0) {
        fsnotify_modify(file);    /* 文件修改事件通知 */
        add_wchar(current, ret);  /* 增加进程写字节计数 */
    }

    /* 增加一次写系统调用计数（per-task 级统计） */
    inc_syscw(current);

    /* 结束文件系统写事务（可能提交日志、释放锁等） */
    file_end_write(file);

    return ret;
}

```


#### 3.4. new_sync_write


```C
/*
 * new_sync_write - 同步写入包装层，目的是同步接口，解决write的同步性
 *
 * 将传统的 (buf, len) 写入参数封装成 kiocb + iov_iter，
 * 然后统一调用 file->f_op->write_iter()。
 *
 * 注意：这是现代 Linux 内核的标准同步写路径，
 *       所有常规文件、socket、pipe、设备等写入最终都会走到这里。
 */
static ssize_t new_sync_write(struct file *filp,
                              const char __user *buf,
                              size_t len,
                              loff_t *ppos)
{
    struct kiocb kiocb;  /* 内核 I/O 控制块，描述一次 IO 操作 */
    struct iov_iter iter; /* 迭代器，用于描述用户缓冲区 */
    ssize_t ret;

    /* 初始化同步 IO 控制块，将 kiocb 与 file 绑定 */
    init_sync_kiocb(&kiocb, filp);

    /* 设置写入的初始偏移（来自 ppos 或 file->f_pos） */
    kiocb.ki_pos = (ppos ? *ppos : 0);

    /* 将用户缓冲区 buf 封装成一个 iov_iter 迭代器 */
    iov_iter_ubuf(&iter, ITER_SOURCE,
                  (void __user *)buf, len);

    /*
     * 核心调用：
     * 调用文件的 write_iter 回调（现代写接口）
     * 几乎所有类型文件系统与驱动都实现了 write_iter。
     */
    ret = filp->f_op->write_iter(&kiocb, &iter);        // 调用具体文件系统的 write/write_iter

    /*
     * -EIOCBQUEUED 表示异步 I/O 队列，但此函数是同步写接口，
     * 因此出现这种情况说明驱动实现错误。
     */
    BUG_ON(ret == -EIOCBQUEUED);

    /* 若写入成功且有有效 ppos，则更新文件偏移 */
    if (ret > 0 && ppos)
        *ppos = kiocb.ki_pos;
    return ret;
}
```

write_iter 是文件系统 / 驱动在 file_operations 中提供的回调函数指针，在fs.h中定义，各种文件类型自己实现。


| 文件类型                | write_iter 实现在哪里？                                    |
| ------------------- | ---------------------------------------------------- |
| **普通文件（ext4）**      | fs/ext4/file.c → `ext4_file_write_iter()`            |
| **XFS 文件系统**        | fs/xfs/xfs_file.c → `xfs_file_write_iter()`          |
| **Btrfs 文件系统**      | fs/btrfs/file.c → `btrfs_file_write_iter()`          |
| **管道 pipe**         | pipe 走的是 `pipe_write()`（不使用 write_iter）              |
| **socket（TCP/UDP）** | net/socket.c → `sock_write_iter()`                   |
| **字符设备（设备驱动）**      | 驱动自己实现其 write()，通常不使用 write_iter                     |
| **终端 tty**          | drivers/tty/tty_io.c → `tty_write()` |

```c
static ssize_t file_tty_write(struct file *file, struct kiocb *iocb, struct iov_iter *from)
{
    struct tty_struct *tty = file_tty(file);
    struct tty_ldisc *ld;
    ssize_t ret;

    /* ① 检查 TTY 结构体是否损坏或非法
     * tty_paranoia_check 是调试用的安全检查，避免空指针或野指针。
     */
    if (tty_paranoia_check(tty, file_inode(file), "tty_write"))
        return -EIO;

    /* ② 基本可写性检查：
     * - tty 是否存在
     * - tty 是否有 ops->write（即底层驱动必须提供 write_room/write）
     * - tty_io_error 是否处于错误状态（比如挂起模式）
     */
    if (!tty || !tty->ops->write || tty_io_error(tty))
        return -EIO;

    /* ③ 检查驱动是否实现 write_room()
     * write_room() 用于检查环形缓冲区或 FIFO 有多少空间
     * 若没有这个函数，属于驱动实现不完整，打印错误日志。
     */
    if (tty->ops->write_room == NULL)
        tty_err(tty, "missing write_room method\n");

    /* ④ 获取行规程对象（ line discipline ）
     * 行规程负责处理高级行为，如：
     * - 回显（echo）
     * - 控制字符（Ctrl-C）
     * - 逐行模式（canonical mode）
     */
    ld = tty_ldisc_ref_wait(tty);

    /* 如果获取失败（TTY 被挂断），转向 hung_up_tty_write */
    if (!ld)
        return hung_up_tty_write(iocb, from);

    /* ⑤ 行规程必须实现 .write（即 n_tty_write）
     * 不实现则报错。
     */
    if (!ld->ops->write)
        ret = -EIO;

    else
        /* ⑥ 实际的写入路径
         * iterate_tty_write 内部会循环调用 ld->ops->write()
         * 并处理 from（iov_iter），直到数据写完。
         *
         * 这个函数是真正把“用户要写的数据”逐块送到行规程的地方。
         */
        ret = iterate_tty_write(ld, tty, file, from);       // 调用行规程的 write

    /* ⑦ 释放行规程引用 */
    tty_ldisc_deref(ld);

    return ret;
}
```

```c
/*
 * 将 write() 的数据拆成合理块大小，避免 DOS 式长数据写入
 * 是对低层驱动的一种保护（降低锁竞争、避免巨量拷贝）
 */
static ssize_t iterate_tty_write(struct tty_ldisc *ld, struct tty_struct *tty,
				 struct file *file, struct iov_iter *from)
{
	size_t chunk, count = iov_iter_count(from);
	ssize_t ret, written = 0;

	/* 加写锁。防止多个线程同时 write 这个 tty。*/
	ret = tty_write_lock(tty, file->f_flags & O_NDELAY);
	if (ret < 0)
		return ret;

	/*
	 * TTY 写入必须分块。
	 *
	 * 分块理由：
	 *   1. 避免直接把巨大 iov_iter 传给驱动（驱动更简单）
	 *   2. 避免行规程（ntty）告诉你它“能处理很多字符”，
	 *      但实际只吃掉一小部分（典型 bug）。
	 *
	 * 默认 chunk = 2KB（NTTY 所能稳定处理的大小）
	 * 若 TTY_NO_WRITE_SPLIT 置位，则允许使用更大的 chunk（最多 64KB）
	 */
	chunk = 2048;
	if (test_bit(TTY_NO_WRITE_SPLIT, &tty->flags))
		chunk = 65536;

	/* 如果剩余数据比 chunk 小，把 chunk 缩小成 count */
	if (count < chunk)
		chunk = count;

	/*
	 * tty->write_buf 是内核为写操作准备的临时缓冲。
	 * 如果容量不够，需要重新分配更大的 buffer。
	 *
	 * write_buf/write_cnt 受 atomic_write_lock 保护，
	 * 确保同时仅有一个写者修改。
	 */
	if (tty->write_cnt < chunk) {
		u8 *buf_chunk;

		/* chunk 小于 1024 就至少扩到 1024 */
		if (chunk < 1024)
			chunk = 1024;

		/* 分配新的写缓冲区 */
		buf_chunk = kvmalloc(chunk, GFP_KERNEL | __GFP_RETRY_MAYFAIL);
		if (!buf_chunk) {
			ret = -ENOMEM;
			goto out;
		}

		/* 替换旧 buffer */
		kvfree(tty->write_buf);
		tty->write_cnt = chunk;
		tty->write_buf = buf_chunk;
	}

	/* 真正开始循环写入 */
	for (;;) {
		size_t size = min(chunk, count);

		/* 从用户 iov_iter 拷贝 size 字节到 tty->write_buf */
		ret = -EFAULT;
		if (copy_from_iter(tty->write_buf, size, from) != size)
			break;

		/*
		 * 核心调用：送给行规程处理
		 *
		 * 对于默认 line discipline = N_TTY，
		 * ld->ops->write == n_tty_write
		 *
		 * N_TTY 会处理：
		 *   - 回显（echo）
		 *   - 行编辑（canonical 模式）
		 *   - 控制字符（Ctrl-C，Ctrl-D，Backspace）
		 *   - input buffer
		 *   - 最终调用 tty->ops->write() 写到驱动
		 */
		ret = ld->ops->write(tty, file, tty->write_buf, size);      
		if (ret <= 0)
			break;

		written += ret;

		/* ldisc 不应该返回 > size，出现则退出（驱动问题）*/
		if (ret > size)
			break;

		/* 如果只吃掉一部分，要把未吃部分放回 iov_iter */
		if (ret != size)
			iov_iter_revert(from, size-ret);

		count -= ret;
		if (!count)
			break;

		/* 若当前进程收到信号（如 Ctrl+C），write 中断 */
		ret = -ERESTARTSYS;
		if (signal_pending(current))
			break;

		/* 主动让出 CPU，避免长时间占用 */
		cond_resched();
	}

	if (written) {
		/* 更新 tty 的活跃时间戳（统计用途） */
		tty_update_time(tty, true);
		ret = written;
	}

out:
	tty_write_unlock(tty);
	return ret;
}
```

```c
static ssize_t n_tty_write(struct tty_struct *tty, struct file *file,
			   const u8 *buf, size_t nr)
{
	const u8 *b = buf;
	DEFINE_WAIT_FUNC(wait, woken_wake_function);
	ssize_t num, retval = 0;

	/* 
	 * 作业控制检查（Job Control）
	 * 如果终端设置 TOSTOP 并且写不是重定向写，
	 * 后台进程写前台终端会被阻止（POSIX 要求）。
	 */
	if (L_TOSTOP(tty) && file->f_op->write_iter != redirected_tty_write) {
		retval = tty_check_change(tty);
		if (retval)
			return retval;	// 通常返回 -EIO 或触发 SIGTTOU
	}

	/* 获取 termios 配置的读锁，禁止写入过程中被修改 */
	down_read(&tty->termios_rwsem);

	/* 
	 * 输出任何待回显字符（比如退格回显等），
	 * 避免这些字符和本次 write 的内容交错导致混乱
	 */
	process_echoes(tty);

	/* 将当前任务加入 TTY 的写等待队列，用于阻塞/唤醒控制 */
	add_wait_queue(&tty->write_wait, &wait);

	while (1) {
		/* 如果收到信号（如 Ctrl-C），中断写操作 */
		if (signal_pending(current)) {
			retval = -ERESTARTSYS;	// 表示写被信号中断
			break;
		}

		/* TTY hangup 检查（设备被挂断或另一端关闭） */
		if (tty_hung_up_p(file) || (tty->link && !tty->link->count)) {
			retval = -EIO;
			break;
		}

		/*
		 * 如果开启 OPOST（输出加工），做字符加工：
		 * 如 \n → \r\n、TAB 展开、控制字符可视化等。
		 * 这是终端层的字符语义处理。
		 */
		if (O_OPOST(tty)) {
			while (nr > 0) {
				/* 批量处理字符（根据 termios 格式转换） */
				num = process_output_block(tty, b, nr);
				if (num < 0) {
					if (num == -EAGAIN)
						break;	// 需要等待驱动可写
					retval = num;	// 其他错误
					goto break_out;
				}

				b += num;
				nr -= num;

				if (nr == 0)
					break;

				/* 对下一个字符做逐字符输出处理 */
				if (process_output(*b, tty) < 0)
					break;
				b++; nr--;
			}

			/* 一些驱动需要 flush_chars 来触发发送（例如 UART） */
			if (tty->ops->flush_chars)
				tty->ops->flush_chars(tty);

		} else {
			/*
			 * 如果关闭 OPOST（Raw 模式），
			 * 直接把数据写入驱动层，不作字符加工。
			 */
			struct n_tty_data *ldata = tty->disc_data;

			while (nr > 0) {
				mutex_lock(&ldata->output_lock);
				/* 调用驱动的 write 函数 */
				num = tty->ops->write(tty, b, nr);
				mutex_unlock(&ldata->output_lock);

				if (num < 0) {
					retval = num;
					goto break_out;	// 驱动错误
				}
				if (!num)
					break;	// 驱动暂时不可写

				b += num;
				nr -= num;
			}
		}

		/* 全部写完，退出循环 */
		if (!nr)
			break;

		/*
		 * 如果文件是 O_NONBLOCK 打开，不允许阻塞等待，
		 * 于是立即返回 -EAGAIN。
		 */
		if (tty_io_nonblock(tty, file)) {
			retval = -EAGAIN;
			break;
		}

		/* 进入休眠前释放 termios 锁 */
		up_read(&tty->termios_rwsem);

		/* 
		 * 挂起等待驱动写缓冲唤醒：
		 * 被 tty_write_wakeup() 等回调唤醒
		 */
		wait_woken(&wait, TASK_INTERRUPTIBLE, MAX_SCHEDULE_TIMEOUT);

		/* 醒来后继续重新拿 termios 锁 */
		down_read(&tty->termios_rwsem);
	}

break_out:
	/* 从等待队列移除 */
	remove_wait_queue(&tty->write_wait, &wait);

	/* 
	 * 如果仍有未写完的数据并设置了 fasync，
	 * 通知用户态异步事件（SIGIO）
	 */
	if (nr && tty->fasync)
		set_bit(TTY_DO_WRITE_WAKEUP, &tty->flags);

	/* 释放 termios 锁 */
	up_read(&tty->termios_rwsem);

	/* 返回已写字节数（优先），否则返回错误码 */
	return (b - buf) ? b - buf : retval;
}
```

```c
static ssize_t acm_tty_write(struct tty_struct *tty, const u8 *buf,
			     size_t count)
{
	struct acm *acm = tty->driver_data;
	int stat;
	unsigned long flags;
	int wbn;
	struct acm_wb *wb;

	/* 如果没有要写的数据，直接返回 */
	if (!count)
		return 0;

	/* 打印调试信息：从 tty 层收到多少字节 */
	dev_vdbg(&acm->data->dev, "%zu bytes from tty layer\n", count);

	/*
	 * 加锁保护 acm->wb[]（写缓冲管理）
	 * 这个锁保护写队列、预防竞争条件（尤其在中断环境）
	 */
	spin_lock_irqsave(&acm->write_lock, flags);

	/*
	 * 从写缓冲池中分配一个 write buffer（acm_wb）
	 * acm_wb_alloc 会找一个尚未使用的 wb slot
	 */
	wbn = acm_wb_alloc(acm);
	if (wbn < 0) {
		/* 没有可用 buffer（一般是写太快 or 正在传输） */
		spin_unlock_irqrestore(&acm->write_lock, flags);
		return 0;   /* 上层会认为暂时写不进去 */
	}
	wb = &acm->wb[wbn];

	/* 如果 USB 设备已经消失或被拔掉，则返回错误 */
	if (!acm->dev) {
		wb->use = false;   /* 释放 buffer slot */
		spin_unlock_irqrestore(&acm->write_lock, flags);
		return -ENODEV;
	}

	/*
	 * 限制一次写入的最大长度
	 * CDC-ACM 每个 URB 的 writesize 是固定的（设备报告）
	 */
	count = (count > acm->writesize) ? acm->writesize : count;
	dev_vdbg(&acm->data->dev, "writing %zu bytes\n", count);

	/* 把用户数据复制到 acm write buffer 缓冲区 */
	memcpy(wb->buf, buf, count);
	wb->len = count;

	/*
	 * USB 自动电源管理：确保 interface 不在 autosuspend 状态。
	 * 异步获取 PM 引用计数，使设备保持唤醒
	 */
	stat = usb_autopm_get_interface_async(acm->control);
	if (stat) {
		/* 如果 PM 获取失败，释放缓冲，返回错误 */
		wb->use = false;
		spin_unlock_irqrestore(&acm->write_lock, flags);
		return stat;
	}

	/*
	 * 如果设备处于 "suspend"（USB 挂起）状态，
	 * 就把 URB 放入 acm->delayed 队列，等待恢复后发送。
	 */
	if (acm->susp_count) {
		usb_anchor_urb(wb->urb, &acm->delayed);
		spin_unlock_irqrestore(&acm->write_lock, flags);
		return count;
	}

	/*
	 * 真正开始发送：提交写 URB 到 USB 栈。
	 * 这一步会让 USB 控制器（如 xHCI）开始 DMA 传输
	 */
	stat = acm_start_wb(acm, wb);                       // acm_start_wb() 调用
	spin_unlock_irqrestore(&acm->write_lock, flags);

	/*
	 * 如果提交失败，返回错误；成功则返回已写入的 count
	 */
	if (stat < 0)
		return stat;

	return count;
}

/*
 * Poke write.
 *
 * 调用者需要提前加锁（即 acm_write_lock 已经持有）
 */

static int acm_start_wb(struct acm *acm, struct acm_wb *wb)
{
	int rc;

	/* 
	 * 标记：当前有一个新的传输正在进行
	 * transmitting 用于防止同时启动多个 URB
	 */
	acm->transmitting++;

	/*
	 * 设置 URB 的发送内容和长度
	 * 每个 wb 预先分配了 urb 和 DMA buffer
	 */
	wb->urb->transfer_buffer = wb->buf;   // 数据起始地址
	wb->urb->transfer_dma = wb->dmah;     // DMA 地址（硬件使用）
	wb->urb->transfer_buffer_length = wb->len;  // 要发送的字节数
	wb->urb->dev = acm->dev;              // 对应的 USB 设备结构体

	/*
	 * 真正提交 USB 传输
	 * 这一步会进入 USB core → xHCI → DMA → 发给设备
	 */
	rc = usb_submit_urb(wb->urb, GFP_ATOMIC);           // usb_submit_urb() 调用,这边进入USB子系统

	if (rc < 0) {
		/* -EPERM = urb 被杀死(如拔掉设备)，属于正常情况 */
		if (rc != -EPERM)
			dev_err(&acm->data->dev,
				"%s - usb_submit_urb(write bulk) failed: %d\n",
				__func__, rc);

		/*
		 * 如果提交失败，需要清理 wb 状态：
		 * - 解锁 wb
		 * - transmitting--
		 * - 释放 buffer 等
		 */
		acm_write_done(acm, wb);
	}

	return rc;
}
```

```c
int usb_submit_urb(struct urb *urb, gfp_t mem_flags)
{
	int				xfertype, max;
	struct usb_device		*dev;
	struct usb_host_endpoint	*ep;
	int				is_out;
	unsigned int			allowed;

	/*--------------------------------------------------------------
	 * 基础合法性检查
	 *--------------------------------------------------------------*/

	/* urb 必须存在，并且必须有 complete 回调，否则完成时没人接收结果 */
	if (!urb || !urb->complete)
		return -EINVAL;

	/* urb->hcpriv 非空 = urb 仍在 host controller 内活跃，不能重复提交 */
	if (urb->hcpriv) {
		WARN_ONCE(1, "URB %pK submitted while active\n", urb);
		return -EBUSY;
	}

	dev = urb->dev;

	/* USB 设备必须存在且至少已通过认证，否则不能传输数据 */
	if ((!dev) || (dev->state < USB_STATE_UNAUTHENTICATED))
		return -ENODEV;

	/*--------------------------------------------------------------
	 * 根据 urb->pipe 找到 endpoint 结构体
	 *--------------------------------------------------------------*/

	ep = usb_pipe_endpoint(dev, urb->pipe);
	if (!ep)
		return -ENOENT;

	/* 把 endpoint 记录到 urb->ep 中，HCD 后面会用到 */
	urb->ep = ep;

	/* 初始化 urb 状态 */
	urb->status = -EINPROGRESS;	/* 代表传输正在进行 */
	urb->actual_length = 0;		/* 实际传输长度清零 */

	/*--------------------------------------------------------------
	 * 控制传输的特殊检查，否则直接根据 endpoint 判断方向
	 *--------------------------------------------------------------*/

	xfertype = usb_endpoint_type(&ep->desc);

	if (xfertype == USB_ENDPOINT_XFER_CONTROL) {
		struct usb_ctrlrequest *setup =
				(struct usb_ctrlrequest *) urb->setup_packet;

		if (!setup)
			return -ENOEXEC;

		/* 控制传输方向由 bRequestType 决定 */
		is_out = !(setup->bRequestType & USB_DIR_IN) ||
				!setup->wLength;

		/* 驱动必须保证 setup->wLength == transfer_buffer_length */
		dev_WARN_ONCE(&dev->dev,
			(usb_pipeout(urb->pipe) != is_out),
			"BOGUS control dir, pipe %x doesn't match bRequestType %x\n",
			urb->pipe, setup->bRequestType);

		if (le16_to_cpu(setup->wLength) != urb->transfer_buffer_length) {
			dev_dbg(&dev->dev,
				"BOGUS control len %d doesn't match transfer length %d\n",
				le16_to_cpu(setup->wLength),
				urb->transfer_buffer_length);
			return -EBADR;
		}

	} else {
		/* 非 control：方向由 endpoint 描述符决定 */
		is_out = usb_endpoint_dir_out(&ep->desc);
	}

	/*--------------------------------------------------------------
	 * 清除旧 flag，写入方向标志，同时为 KMSAN 做标记
	 *--------------------------------------------------------------*/

	urb->transfer_flags &= ~(URB_DIR_MASK | URB_DMA_MAP_SINGLE |
			URB_DMA_MAP_PAGE | URB_DMA_MAP_SG | URB_MAP_LOCAL |
			URB_SETUP_MAP_SINGLE | URB_SETUP_MAP_LOCAL |
			URB_DMA_SG_COMBINED);

	/* 写入方向：URB_DIR_OUT 或 URB_DIR_IN */
	urb->transfer_flags |= (is_out ? URB_DIR_OUT : URB_DIR_IN);

	kmsan_handle_urb(urb, is_out);

	/*--------------------------------------------------------------
	 * 除 control 端点外，设备必须处在 CONFIGURED 状态
	 *--------------------------------------------------------------*/

	if (xfertype != USB_ENDPOINT_XFER_CONTROL &&
			dev->state < USB_STATE_CONFIGURED)
		return -ENODEV;

	/*--------------------------------------------------------------
	 * 端点最大包长检查（MaxPacketSize）
	 *--------------------------------------------------------------*/

	max = usb_endpoint_maxp(&ep->desc);
	if (max <= 0) {
		dev_dbg(&dev->dev,
			"bogus endpoint ep%d%s in %s (bad maxpacket %d)\n",
			usb_endpoint_num(&ep->desc), is_out ? "out" : "in",
			__func__, max);
		return -EMSGSIZE;
	}

	/*--------------------------------------------------------------
	 * Isochronous 端点的特殊校验
	 *--------------------------------------------------------------*/

	if (xfertype == USB_ENDPOINT_XFER_ISOC) {
		int	n, len;

		/* SuperSpeed 以上的 iso 端点必须考虑 burst/mult */
		if (dev->speed >= USB_SPEED_SUPER) {
			int burst = 1 + ep->ss_ep_comp.bMaxBurst;
			int mult = USB_SS_MULT(ep->ss_ep_comp.bmAttributes);
			max *= burst;
			max *= mult;
		}

		/* SuperSpeedPlus 单独字段 */
		if (dev->speed == USB_SPEED_SUPER_PLUS &&
		    USB_SS_SSP_ISOC_COMP(ep->ss_ep_comp.bmAttributes)) {
			struct usb_ssp_isoc_ep_comp_descriptor *isoc_ep_comp;

			isoc_ep_comp = &ep->ssp_isoc_ep_comp;
			max = le32_to_cpu(isoc_ep_comp->dwBytesPerInterval);
		}

		/* 高速 Isochronous 的多包模式 */
		if (dev->speed == USB_SPEED_HIGH)
			max *= usb_endpoint_maxp_mult(&ep->desc);

		/* 每个 iso_frame_desc 都要检查 */
		if (urb->number_of_packets <= 0)
			return -EINVAL;

		for (n = 0; n < urb->number_of_packets; n++) {
			len = urb->iso_frame_desc[n].length;

			if (len < 0 || len > max)
				return -EMSGSIZE;

			urb->iso_frame_desc[n].status = -EXDEV;
			urb->iso_frame_desc[n].actual_length = 0;
		}

	/*--------------------------------------------------------------
	 * SG（scatter-gather）模式下的对齐检查
	 *--------------------------------------------------------------*/
	} else if (urb->num_sgs && !urb->dev->bus->no_sg_constraint) {
		struct scatterlist *sg;
		int i;

		for_each_sg(urb->sg, sg, urb->num_sgs - 1, i)
			if (sg->length % max)
				return -EINVAL;
	}

	/* 长度超过 INT_MAX 防止内核整数溢出 */
	if (urb->transfer_buffer_length > INT_MAX)
		return -EMSGSIZE;

	/*--------------------------------------------------------------
	 * 警告：pipe 的类型必须与端点类型兼容
	 *--------------------------------------------------------------*/

	if (usb_pipe_type_check(urb->dev, urb->pipe))
		dev_warn_once(&dev->dev,
			"BOGUS urb xfer, pipe %x != type %x\n",
			usb_pipetype(urb->pipe), pipetypes[xfertype]);

	/*--------------------------------------------------------------
	 * 根据不同传输类型过滤合法的 transfer_flags
	 *--------------------------------------------------------------*/

	allowed = (URB_NO_TRANSFER_DMA_MAP | URB_NO_INTERRUPT |
			URB_DIR_MASK | URB_FREE_BUFFER);

	switch (xfertype) {
	case USB_ENDPOINT_XFER_BULK:
	case USB_ENDPOINT_XFER_INT:
		if (is_out)
			allowed |= URB_ZERO_PACKET;
		fallthrough;

	default:	/* 除 iso 外所有端点类型 */
		if (!is_out)
			allowed |= URB_SHORT_NOT_OK;
		break;

	case USB_ENDPOINT_XFER_ISOC:
		allowed |= URB_ISO_ASAP;
		break;
	}

	allowed &= urb->transfer_flags;

	if (allowed != urb->transfer_flags)
		dev_WARN(&dev->dev, "BOGUS urb flags, %x --> %x\n",
			urb->transfer_flags, allowed);

	/*--------------------------------------------------------------
	 * 中断 / Iso 端点的 interval（周期）合法化
	 *--------------------------------------------------------------*/

	switch (xfertype) {
	case USB_ENDPOINT_XFER_ISOC:
	case USB_ENDPOINT_XFER_INT:

		if (urb->interval <= 0)
			return -EINVAL;

		switch (dev->speed) {

		case USB_SPEED_SUPER_PLUS:
		case USB_SPEED_SUPER:	/* 125us 一个微帧 */
			if (urb->interval > (1 << 15))
				return -EINVAL;
			max = 1 << 15;
			break;

		case USB_SPEED_HIGH:	/* 微帧 */
			if (urb->interval > (1024 * 8))
				urb->interval = 1024 * 8;
			max = 1024 * 8;
			break;

		case USB_SPEED_FULL:	/* 1ms 帧 */
		case USB_SPEED_LOW:
			if (xfertype == USB_ENDPOINT_XFER_INT) {
				if (urb->interval > 255)
					return -EINVAL;
				max = 128; /* OHCI 限制更严格 */
			} else {
				if (urb->interval > 1024)
					urb->interval = 1024;
				max = 1024;
			}
			break;

		default:
			return -EINVAL;
		}

		/* interval 必须是 2 的幂，向下规整 */
		urb->interval = min(max, 1 << ilog2(urb->interval));
	}

	/*--------------------------------------------------------------
	 * 最终调用：交给 Host Controller Driver 处理
	 *--------------------------------------------------------------*/

	return usb_hcd_submit_urb(urb, mem_flags);
}
```

```c
int usb_hcd_submit_urb (struct urb *urb, gfp_t mem_flags)
{
	int			status;
	struct usb_hcd		*hcd = bus_to_hcd(urb->dev->bus);

	/*--------------------------------------------------------------
	 * 【块 1：初始化与引用计数】
	 *
	 * USB Core 在把 URB 交给 HCD（硬件控制器驱动）之前必须确保：
	 * - 增加 URB 自身引用计数
	 * - 增加 URB 使用计数（use_count）
	 * - 增加设备 URB 数计数（urbnum）
	 * 同时记录到 usbmon，以便调试。
	 *--------------------------------------------------------------*/
	usb_get_urb(urb);
	atomic_inc(&urb->use_count);
	atomic_inc(&urb->dev->urbnum);
	usbmon_urb_submit(&hcd->self, urb);


	/*--------------------------------------------------------------
	 * 【块 2：区分 Root Hub / 普通 USB 设备】
	 *
	 * Root Hub（虚拟 hub）不走真实硬件控制器，因此直接调
	 * rh_urb_enqueue()，由内核软件模拟响应。
	 *
	 * 普通设备：
	 *   1. 执行 DMA 映射（map_urb_for_dma）
	 *   2. 调用 HCD 驱动的 urb_enqueue（如 DWC2/XHCI/EHCI）
	 *   3. 若失败则取消 DMA 映射
	 *--------------------------------------------------------------*/
	if (is_root_hub(urb->dev)) {
		status = rh_urb_enqueue(hcd, urb);

	} else {
		status = map_urb_for_dma(hcd, urb, mem_flags);

		if (likely(status == 0)) {
			status = hcd->driver->urb_enqueue(hcd, urb, mem_flags);     //核心就这么一句USB Core → 具体硬件控制器（如 DWC2 / DWC3 / XHCI / EHCI）

			if (unlikely(status))
				unmap_urb_for_dma(hcd, urb);
		}
	}


	/*--------------------------------------------------------------
	 * 【块 3：错误处理路径】
	 *
	 * 如果提交失败：
	 *   - 记录 usbmon 错误事件
	 *   - 清理 hcpriv 与 urb_list
	 *   - 回收 use_count 与 urbnum
	 *   - 若 URB 在提交中被 kill/poison，唤醒等待队列
	 *   - 释放 URB（usb_put_urb）
	 *
	 * 成功则直接返回 0。
	 *--------------------------------------------------------------*/
	if (unlikely(status)) {
		usbmon_urb_submit_error(&hcd->self, urb, status);

		urb->hcpriv = NULL;
		INIT_LIST_HEAD(&urb->urb_list);

		atomic_dec(&urb->use_count);
		smp_mb__after_atomic();

		atomic_dec(&urb->dev->urbnum);

		if (atomic_read(&urb->reject))
			wake_up(&usb_kill_urb_queue);

		usb_put_urb(urb);
	}

	return status;
}
```

```c
/*--------------------------------------------------------------
 * xhci_urb_enqueue()
 *
 * 功能：将一个 URB 放入 xHCI（USB3 控制器）的传输队列。
 *       根据端点类型选择不同的 queue_xxx_tx() 函数，
 *       并为本次 URB 分配对应的传输描述（TD，多个 TRB）。
 *
 * write() 调用 cdc_acm_tty_write → usb_submit_urb →
 * usb_hcd_submit_urb →（这里）
 *
 * 这是 USB3 主机控制器处理 URB 的“入口函数”。
 *--------------------------------------------------------------*/
static int xhci_urb_enqueue(struct usb_hcd *hcd, struct urb *urb, gfp_t mem_flags)
{
	struct xhci_hcd *xhci = hcd_to_xhci(hcd);
	unsigned long flags;
	int ret = 0;
	unsigned int slot_id, ep_index;
	unsigned int *ep_state;
	struct urb_priv	*urb_priv;
	int num_tds;

	/*--------------------------------------------------------------
	 * 块 1：计算此 URB 需要多少个 TD（Transfer Descriptor）
	 *
	 * xHCI 中一个 TD 对应一个或多个 TRB：
	 *   - Isochronous（视频）：每个 packet 一个 TD
	 *   - Bulk OUT（串口写）：通常 1 个 TD（必要时 2 个）
	 *   - Control：按协议划分
	 *--------------------------------------------------------------*/
	ep_index = xhci_get_endpoint_index(&urb->ep->desc);

	if (usb_endpoint_xfer_isoc(&urb->ep->desc))
		num_tds = urb->number_of_packets;
	else if (usb_endpoint_is_bulk_out(&urb->ep->desc) &&
	    urb->transfer_buffer_length > 0 &&
	    urb->transfer_flags & URB_ZERO_PACKET &&
	    !(urb->transfer_buffer_length % usb_endpoint_maxp(&urb->ep->desc)))
		num_tds = 2;			/* 需要发送 Zero-Length Packet */
	else
		num_tds = 1;

	/*--------------------------------------------------------------
	 * 块 2：分配 URB 的私有数据（urb_priv）
	 * 用于保存本次提交产生的 TD/TRB 信息
	 *--------------------------------------------------------------*/
	urb_priv = kzalloc(struct_size(urb_priv, td, num_tds), mem_flags);
	if (!urb_priv)
		return -ENOMEM;

	urb_priv->num_tds = num_tds;
	urb_priv->num_tds_done = 0;
	urb->hcpriv = urb_priv;

	trace_xhci_urb_enqueue(urb);

	/*--------------------------------------------------------------
	 * 块 3：加锁并检查 xHCI 状态与 EP 状态
	 *   - 主控制器是否可访问
	 *   - 端口是否错误
	 *   - 设备 slot 是否有效
	 *   - 端点是否处于异常状态（streams 或 toggle 清理）
	 *--------------------------------------------------------------*/
	spin_lock_irqsave(&xhci->lock, flags);

	ret = xhci_check_args(hcd, urb->dev, urb->ep,
			      true, true, __func__);
	if (ret <= 0) {
		ret = ret ? ret : -EINVAL;
		goto free_priv;
	}

	slot_id = urb->dev->slot_id;

	if (!HCD_HW_ACCESSIBLE(hcd)) {
		ret = -ESHUTDOWN;
		goto free_priv;
	}

	if (xhci->devs[slot_id]->flags & VDEV_PORT_ERROR) {
		xhci_dbg(xhci, "Can't queue urb, port error, link inactive\n");
		ret = -ENODEV;
		goto free_priv;
	}

	if (xhci->xhc_state & XHCI_STATE_DYING) {
		xhci_dbg(xhci, "Ep 0x%x: URB %p submitted for non-responsive xHCI host.\n",
			 urb->ep->desc.bEndpointAddress, urb);
		ret = -ESHUTDOWN;
		goto free_priv;
	}

	ep_state = &xhci->devs[slot_id]->eps[ep_index].ep_state;

	if (*ep_state & (EP_GETTING_STREAMS | EP_GETTING_NO_STREAMS)) {
		xhci_warn(xhci, "WARN: Can't enqueue URB, ep in streams transition state %x\n",
			  *ep_state);
		ret = -EINVAL;
		goto free_priv;
	}
	if (*ep_state & EP_SOFT_CLEAR_TOGGLE) {
		xhci_warn(xhci, "Can't enqueue URB while manually clearing toggle\n");
		ret = -EINVAL;
		goto free_priv;
	}

	/*--------------------------------------------------------------
	 * 块 4：根据端点类型选择 TX 函数
	 *
	 * 这里正式进入 xHCI 的 TRB 构造逻辑：
	 *   Control → xhci_queue_ctrl_tx()			控制传输
	 *   Bulk   → xhci_queue_bulk_tx()			批量传输，写字符控制肯定是走的这个
	 *   Intr   → xhci_queue_intr_tx()			中断传输
	 *   Isoc   → xhci_queue_isoc_tx_prepare()	等时传输
	 *
	 * 对于你的 USB 串口（cdc_acm）write：
	 *   → 这是 Bulk OUT
	 *   → 会进入 xhci_queue_bulk_tx()
	 *--------------------------------------------------------------*/
	switch (usb_endpoint_type(&urb->ep->desc)) {

	case USB_ENDPOINT_XFER_CONTROL:
		ret = xhci_queue_ctrl_tx(xhci, GFP_ATOMIC, urb,
					 slot_id, ep_index);		
		break;

	case USB_ENDPOINT_XFER_BULK:
		ret = xhci_queue_bulk_tx(xhci, GFP_ATOMIC, urb,
					 slot_id, ep_index);		// 串口写
		break;

	case USB_ENDPOINT_XFER_INT:
		ret = xhci_queue_intr_tx(xhci, GFP_ATOMIC, urb,
				slot_id, ep_index);
		break;

	case USB_ENDPOINT_XFER_ISOC:
		ret = xhci_queue_isoc_tx_prepare(xhci, GFP_ATOMIC, urb,
				slot_id, ep_index);
	}

	/*--------------------------------------------------------------
	 * 块 5：错误处理与解锁
	 * 如果队列失败，释放 urb_priv，撤销本次提交
	 *--------------------------------------------------------------*/
	if (ret) {
free_priv:
		xhci_urb_free_priv(urb_priv);
		urb->hcpriv = NULL;
	}

	spin_unlock_irqrestore(&xhci->lock, flags);
	return ret;
}


/*
 * xhci_queue_bulk_tx()
 *
 * 功能：为一个 Bulk URB 构造 xHCI 所需的一系列 NORMAL TRB，
 *       把它们排入对应的 Transfer Ring，并在最后交给硬件。
 *       这是 USB3 Bulk 传输（比如你写串口数据）的核心数据路径。
 */
int xhci_queue_bulk_tx(struct xhci_hcd *xhci, gfp_t mem_flags,
		struct urb *urb, int slot_id, unsigned int ep_index)
{
	struct xhci_ring *ring;
	struct urb_priv *urb_priv;
	struct xhci_td *td;
	struct xhci_generic_trb *start_trb;
	struct scatterlist *sg = NULL;
	bool more_trbs_coming = true;
	bool need_zero_pkt = false;
	bool first_trb = true;
	unsigned int num_trbs;
	unsigned int start_cycle, num_sgs = 0;
	unsigned int enqd_len, block_len, trb_buff_len, full_len;
	int sent_len, ret;
	u32 field, length_field, remainder;
	u64 addr, send_addr;

	/*--------------------------------------------------------------
	 * 块 1：找到该 URB 对应的 Transfer Ring，并决定需要多少个 TRB
	 *
	 * - 根据端点与 stream 选出 ring
	 * - 如果使用 scatter/gather，则从 SG list 中取物理地址与长度
	 * - 计算需要多少 TRB（可能多个 TRB 才能覆盖整个传输）
	 * - 调用 prepare_transfer() 为本次传输分配/初始化 TD、ring 资源
	 *--------------------------------------------------------------*/
	ring = xhci_urb_to_transfer_ring(xhci, urb);
	if (!ring)
		return -EINVAL;

	full_len = urb->transfer_buffer_length;

	/* 如果有 scatter/gather list，就使用 SG 模式，否则用单一缓冲区 */
	if (urb->num_sgs && !(urb->transfer_flags & URB_DMA_MAP_SINGLE)) {
		num_sgs = urb->num_mapped_sgs;
		sg = urb->sg;
		addr = (u64) sg_dma_address(sg);
		block_len = sg_dma_len(sg);
		num_trbs = count_sg_trbs_needed(urb);
	} else {
		num_trbs = count_trbs_needed(urb);
		addr = (u64) urb->transfer_dma;
		block_len = full_len;
	}

	ret = prepare_transfer(xhci, xhci->devs[slot_id],
			ep_index, urb->stream_id,
			num_trbs, urb, 0, mem_flags);
	if (unlikely(ret < 0))
		return ret;

	urb_priv = urb->hcpriv;

	/*--------------------------------------------------------------
	 * 块 2：处理 URB_ZERO_PACKET 语义
	 *
	 * 若设置了 URB_ZERO_PACKET 且有多于 1 个 TD，说明需要在
	 * 数据末尾再补一个 0 长度包（常用于 bulk out 满包时强制结束）。
	 *--------------------------------------------------------------*/
	if (urb->transfer_flags & URB_ZERO_PACKET && urb_priv->num_tds > 1)
		need_zero_pkt = true;

	td = &urb_priv->td[0];

	/*--------------------------------------------------------------
	 * 块 3：记录首个 TRB 位置和当前 cycle state
	 *
	 * 为了避免在构建整个 TD 之前就“激活”首个 TRB，这里：
	 * - 保存 ring->enqueue 指针（start_trb）
	 * - 保存当前 cycle_state，用于之后统一处理首 TRB 的 CYCLE 位
	 *--------------------------------------------------------------*/
	start_trb = &ring->enqueue->generic;
	start_cycle = ring->cycle_state;
	send_addr = addr;

	/*--------------------------------------------------------------
	 * 块 4：核心循环——将整个缓冲区切分为若干 TRB 并入队
	 *
	 * 对整段数据（full_len）反复：
	 *   - 计算本次 TRB 能承载的长度（不能跨 64KB 边界）
	 *   - 设置 TRB 的控制字段（CYCLE/CHAIN/IOC/IDT/ISP 等）
	 *   - 计算 TD 剩余包数 remainder 并写入长度字段
	 *   - 调用 queue_trb() 把 TRB 写入 ring
	 *   - 若使用 SG，按 SG 边界推进
	 *--------------------------------------------------------------*/
	for (enqd_len = 0; first_trb || enqd_len < full_len;
			enqd_len += trb_buff_len) {
		field = TRB_TYPE(TRB_NORMAL);

		/* TRB buffer 不允许跨 64KB 边界，先裁剪到边界内 */
		trb_buff_len = TRB_BUFF_LEN_UP_TO_BOUNDARY(addr);
		trb_buff_len = min_t(unsigned int, trb_buff_len, block_len);

		/* 再裁剪到剩余总长度范围内 */
		if (enqd_len + trb_buff_len > full_len)
			trb_buff_len = full_len - enqd_len;

		/* 首个 TRB 的 cycle 位延后处理，避免提前交给硬件 */
		if (first_trb) {
			first_trb = false;
			if (start_cycle == 0)
				field |= TRB_CYCLE;
		} else {
			field |= ring->cycle_state;
		}

		/*------------------------------------------------------
		 * CHAIN 位：将多个 TRB 链接为一个 TD
		 * - 非最后一个 TRB：设置 TRB_CHAIN
		 * - 最后一个 TRB：清除 TRB_CHAIN，设置 TRB_IOC
		 *   表示 TD 结束并在完成时产生中断
		 *
		 * 同时考虑 link TRB 对齐需求，需要时调用 xhci_align_td()
		 *------------------------------------------------------*/
		if (enqd_len + trb_buff_len < full_len) {
			field |= TRB_CHAIN;
			if (trb_is_link(ring->enqueue + 1)) {
				if (xhci_align_td(xhci, urb, enqd_len,
						  &trb_buff_len,
						  ring->enq_seg)) {
					send_addr = ring->enq_seg->bounce_dma;
					/* 假定一个 TD 不会跨两个 segment */
					td->bounce_seg = ring->enq_seg;
				}
			}
		}

		if (enqd_len + trb_buff_len >= full_len) {
			/* 最后一个 TRB：去掉 CHAIN，设置 IOC */
			field &= ~TRB_CHAIN;
			field |= TRB_IOC;
			more_trbs_coming = false;
			td->last_trb = ring->enqueue;
			td->last_trb_seg = ring->enq_seg;

			/* 如 URB 适合做 IDT（Immediate Data TRB），则把数据
			 * 直接放入 TRB 而不是通过 DMA 地址指向
			 */
			if (xhci_urb_suitable_for_idt(urb)) {
				memcpy(&send_addr, urb->transfer_buffer,
				       trb_buff_len);
				le64_to_cpus(&send_addr);
				field |= TRB_IDT;
			}
		}

		/* 仅对 IN 端点设置 ISP（Interrupt on Short Packet） */
		if (usb_urb_dir_in(urb))
			field |= TRB_ISP;

		/*------------------------------------------------------
		 * 设置 TRB 长度与 TD 尺寸字段
		 *
		 * remainder = 该 TD 还剩多少包/字节未完成，用于硬件做
		 * short packet 判断等逻辑。
		 *------------------------------------------------------*/
		remainder = xhci_td_remainder(xhci, enqd_len, trb_buff_len,
					      full_len, urb, more_trbs_coming);

		length_field = TRB_LEN(trb_buff_len) |
			TRB_TD_SIZE(remainder) |
			TRB_INTR_TARGET(0);

		/* 真正把 TRB 写入 ring，移动 enqueue 指针 */	// 这里是核心写操作执行的地方
		queue_trb(xhci, ring, more_trbs_coming | need_zero_pkt,
				lower_32_bits(send_addr),
				upper_32_bits(send_addr),
				length_field,
				field);									

		addr += trb_buff_len;
		sent_len = trb_buff_len;

		/*------------------------------------------------------
		 * 若使用 SG：按 sg 边界推进到下一个 sg entry
		 *------------------------------------------------------*/
		while (sg && sent_len >= block_len) {
			/* 切换到下一个 sg */
			--num_sgs;
			sent_len -= block_len;
			sg = sg_next(sg);
			if (num_sgs != 0 && sg) {
				block_len = sg_dma_len(sg);
				addr = (u64) sg_dma_address(sg);
				addr += sent_len;
			}
		}
		block_len -= sent_len;
		send_addr = addr;
	}

	/*--------------------------------------------------------------
	 * 块 5：如需要 Zero-Length Packet，则再补一个 0 长度 TRB
	 *
	 * 某些协议需要在“刚好满包”的情况下再发一个空包来表示结束。
	 * 这里为此单独准备一个额外 TD/ TRB。
	 *--------------------------------------------------------------*/
	if (need_zero_pkt) {
		ret = prepare_transfer(xhci, xhci->devs[slot_id],
				       ep_index, urb->stream_id,
				       1, urb, 1, mem_flags);
		urb_priv->td[1].last_trb = ring->enqueue;
		urb_priv->td[1].last_trb_seg = ring->enq_seg;
		field = TRB_TYPE(TRB_NORMAL) | ring->cycle_state | TRB_IOC;
		queue_trb(xhci, ring, 0, 0, 0, TRB_INTR_TARGET(0), field);
	}

	/*--------------------------------------------------------------
	 * 块 6：检查 TRB/长度一致性、应用硬件 quirks，并“激活”首 TRB
	 *
	 * - check_trb_math()：确认 enqd_len 与 TRB 配置匹配
	 * - 可选地应用某些控制器的特殊 workaround（VL805）
	 * - giveback_first_trb()：在构造完全部 TRB 后，处理首 TRB 的
	 *   cycle 位，让硬件看到完整的 TD
	 *--------------------------------------------------------------*/
	check_trb_math(urb, enqd_len);
	if (xhci->quirks & XHCI_VLI_HUB_TT_QUIRK)
		xhci_vl805_hub_tt_quirk(xhci, urb, ring);

	giveback_first_trb(xhci, slot_id, ep_index, urb->stream_id,
			start_cycle, start_trb);	// 这里面会调用到xhci_ring_doorbell
	return 0;
}

static void queue_trb(struct xhci_hcd *xhci, struct xhci_ring *ring,
		bool more_trbs_coming,
		u32 field1, u32 field2, u32 field3, u32 field4)
{
	struct xhci_generic_trb *trb;

	/*--------------------------------------------------------------
	 * 写入 TRB 字段 0/1/2（地址低、高、长度等）
	 *
	 * 注意：此处是“纯内存写操作”，写入 DMA 可见区域。
	 * queue_trb() 只是“写 DMA 描述符”。
	 * doorbell 触发 xHCI 控制器读这些 TRB，并把数据发到 USB。
	 *--------------------------------------------------------------*/
	trb = &ring->enqueue->generic;
	trb->field[0] = cpu_to_le32(field1);
	trb->field[1] = cpu_to_le32(field2);
	trb->field[2] = cpu_to_le32(field3);


	/* make sure TRB is fully written before giving it to the controller */
	wmb();			// 内存屏障，保证内存写入的顺序
	trb->field[3] = cpu_to_le32(field4);		// field[3] 包含最重要的控制位(CYCLE、CHAIN、IOC等)，xHCI 会以 field[3] 是否匹配当前 cycle bit 来判断 TRB 是否有效。

	trace_xhci_queue_trb(ring, trb);

	inc_enq(xhci, ring, more_trbs_coming);
}



static void giveback_first_trb(struct xhci_hcd *xhci, int slot_id,
		unsigned int ep_index, unsigned int stream_id, int start_cycle,
		struct xhci_generic_trb *start_trb)
{
	/*--------------------------------------------------------------
	 * wmb() 保证所有 TRB 字段已经写入内存（DMA 区域）
	 * 设置 TRB 的 CYCLE bit（C=1 表示 TRB 有效）
	 *
	 * 这是 xHCI 协议中激活 TRB 的关键！
	 * 如果 CYCLE 错误，硬件根本不会处理这个 TRB。
	 *--------------------------------------------------------------*/
	
	wmb();
	if (start_cycle)
		start_trb->field[3] |= cpu_to_le32(start_cycle);
	else
		start_trb->field[3] &= cpu_to_le32(~TRB_CYCLE);
	xhci_ring_ep_doorbell(xhci, slot_id, ep_index, stream_id);	// 这里面会调用到xhci_ring_doorbell
}


/*--------------------------------------------------------------
 * xhci_ring_ep_doorbell()
 *
 * 功能：
 *   - 真正写 doorbell MMIO 寄存器，触发 xHCI 控制器开始读取 TRB ring。
 *   - 加入读写屏障，确保在非一致性系统上（例如 RPi 4/5 的 PCIe）
 *     DMA 和寄存器写入顺序正确。
 *--------------------------------------------------------------*/
void xhci_ring_ep_doorbell(struct xhci_hcd *xhci,
		unsigned int slot_id,
		unsigned int ep_index,
		unsigned int stream_id)
{
	__le32 __iomem *db_addr = &xhci->dba->doorbell[slot_id];
	struct xhci_virt_ep *ep = &xhci->devs[slot_id]->eps[ep_index];
	unsigned int ep_state = ep->ep_state;

	/*--------------------------------------------------------------
	 * 若端点处于清除、停止、迁移等状态，则不能使用 doorbell。
	 *--------------------------------------------------------------*/
	if ((ep_state & EP_STOP_CMD_PENDING) || (ep_state & SET_DEQ_PENDING) ||
	    (ep_state & EP_HALTED) || (ep_state & EP_CLEARING_TT))
		return;

	trace_xhci_ring_ep_doorbell(slot_id, DB_VALUE(ep_index, stream_id));

	/*--------------------------------------------------------------
	 * 为了确保 TRB 的内存写入已经在总线上可见，
	 * 在非一致性系统（例如 Pi 4/5）上，需要读一次 doorbell 寄存器。
	 *
	 * 这强制 CPU → PCIe → xHC DMA 引擎之间的同步。
	 *--------------------------------------------------------------*/
	readl(db_addr);

	/*--------------------------------------------------------------
	 * ★★★★★ 核心：doorbell 写入 ★★★★★
	 *
	 * 写入 doorbell[slot_id] 寄存器，通知 xHCI：
	 *   “对应的 endpoint ring 上有新的 TRB 可以处理了。”
	 *
	 * 写完后 xHCI 硬件开始：
	 *   - DMA 读取 TRB descriptor
	 *   - DMA 读取 URB buffer 数据
	 *   - 发送到 USB 设备（例如你的舵机板）
	 *--------------------------------------------------------------*/
	writel(DB_VALUE(ep_index, stream_id), db_addr);		// 这是 write() 写串口数据时，USB3 控制器真正的Start DMA Transfer，向内存映射的寄存器写一个 32-bit 值


	/* 再读一次确保写入完成（flush） */
	readl(db_addr);
}

```


`writel(DB_VALUE(ep_index, stream_id), db_addr);`这个地址映射的是`DWC3 内部 xHCI doorbell 寄存器`，之后发生的DMA行为：
- 读取 TRB（你 queue_trb 写入的）
- 向 USB PHY 发送包
- 接收 ACK
- 产生 Event TRB 回传
全部由 DWC3 硬件内部执行。
