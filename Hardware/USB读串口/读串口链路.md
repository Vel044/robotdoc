```python
def _sync_read(self, addr: int, length: int, motor_ids: list[int], ...):
    self._setup_sync_reader(motor_ids, addr, length)  # 设置读取参数
    for n_try in range(1 + num_retry):
        comm = self.sync_reader.txRxPacket()  # ← 调用 Feetech SDK 发送并接收数据
        if self._is_comm_success(comm):
            break
    values = {id_: self.sync_reader.getData(id_, addr, length) for id_ in motor_ids}
    return values, comm
```

```python
def rxPacket(self):
    self.last_result = False
    result = COMM_RX_FAIL
    if len(self.data_dict.keys()) == 0:
        return COMM_NOT_AVAILABLE
    for scs_id in self.data_dict:
        self.data_dict[scs_id], result, _ = self.ph.readRx(self.port, scs_id, self.data_length)
        if result != COMM_SUCCESS:
            return result
    if result == COMM_SUCCESS:
        self.last_result = True
    return result
```

```python
def readRx(self, port, scs_id, length):
    result = COMM_TX_FAIL
    error = 0
    rxpacket = None
    data = []
    while True:
        rxpacket, result = self.rxPacket(port)
        if result != COMM_SUCCESS or rxpacket[PKT_ID] == scs_id:
            break
    if result == COMM_SUCCESS and rxpacket[PKT_ID] == scs_id:
        error = rxpacket[PKT_ERROR]
        data.extend(rxpacket[PKT_PARAMETER0: PKT_PARAMETER0 + length])
    return data, result, error
```

```python
def rxPacket(self, port):
    rxpacket = []
    result = COMM_TX_FAIL
    checksum = 0
    rx_length = 0
    wait_length = 6  # minimum length (HEADER0 HEADER1 ID LENGTH ERROR CHKSUM)
    while True:
        rxpacket.extend(port.readPort(wait_length - rx_length))  # ← 读取串口数据
        rx_length = len(rxpacket)
        if rx_length >= wait_length:
            # 查找数据包头
            for idx in range(0, (rx_length - 1)):
                if (rxpacket[idx] == 0xFF) and (rxpacket[idx + 1] == 0xFF):
                    break
            # 检查数据包
            # ...
        else:
            # 检查超时
            if port.isPacketTimeout():
                if rx_length == 0:
                    result = COMM_RX_TIMEOUT
                else:
                    result = COMM_RX_CORRUPT
                break
    port.is_using = False
    return rxpacket, result
```

```python
def read(self, size=1):
    # 调用 select() 等待数据可读
    ready, _, _ = select.select([self.fd], [], [], self.timeout)
    if ready:
        return os.read(self.fd, size)
    return b''
```

```python
def setupPort(self, cflag_baud):

        self.closePort()
    self.ser = serial.Serial(
        port=self.port_name,
        baudrate=self.baudrate,
```

## 调用链时间分布

| 阶段 | 耗时 | 说明 |
|------|------|------|
| 1. LeRobot 应用层 | 几微秒 | 函数调用开销 |
| 2. MotorsBus | 几微秒 | 参数准备 |
| 3. Feetech SDK | 几毫秒 | 协议处理和串口读写 |
| 4. pyserial + select | 主要耗时 | 等待舵机响应 |
| 5. 内核 + 驱动 | 微秒级 | 系统调用和中断处理 |
| 6. 硬件 | 毫秒级 | 舵机处理和数据返回 |

### 优化建议
- **批量读取**：使用 `sync_read` 一次性读取多个电机的位置
- **合理超时**：设置合适的串口超时时间
- **减少频率**：如果不需要每帧都读取舵机位置，可降低读取频率
- **异步读取**：考虑使用异步IO减少阻塞时间
